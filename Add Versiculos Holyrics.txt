function scriptAction(obj) {
    
     // COPIE APENAS VERSÍCULOS
    // Os versículos de livros que possuem o mesmo nome (ex. 1 Samuel e 2 Samuel) devem estar com o numeral ao ser copiado. O uso de "I" ou "II", interrompe a adição dos versículos à playlist!

    // obtém os versículos da área de transferência
    var val = h.getClipboard();
           
    // separa cada versículo em uma linha
    var i = val.split('\n').filter(linha => linha.trim() !== "");;            
    
    //transforma cada linha em um item de versículo
    var items = i.map(linha =>{
        return{
                type: 'verse',
                references: linha,
            };
        });
        
    // adiciona os versículos à Lista de Reprodução de Mídia      
    var r = h.hly('AddToPlaylist', {
    items: items,
          
});
// informa se a área de transferência está vazia ou sem versículos  
if (val === "") {
    h.log("Área de transferência vazia");
} else if (r.status != 'ok') {
    h.log("Área de transferência inválida\n(" + val + ")");
} else {
    h.log(val + "\nVersículos adicionados com sucesso")
}

}
